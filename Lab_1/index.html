Програмування

Я вивчаю мову програмування JavaScript. Ось невеликий приклад програми:

function calculateArea(radius) {
  const PI = 3.14159;
  let area = PI * radius ** 2;
  return area;
}

let r = 5;
let result = calculateArea(r);
console.log('Площа кола з радіусом ' + r + ' дорівнює ' + result);
